\section{Permutation Test}

\begin{frame}{Time Spent: Difference in Means}
    \begin{itemize}
        \item Hypotheses:
        \[
            H_0: \mu_{\text{new}} \le \mu_{\text{old}}, \qquad
            H_1: \mu_{\text{new}} > \mu_{\text{old}}.
        \]

        \item Sample means:
        \[
            \bar{Y}_{\text{old}} \approx 4.53, \qquad
            \bar{Y}_{\text{new}} \approx 6.22.
        \]

        \item Observed statistic:
        \[
            t_{\text{obs}} = \bar{Y}_{\text{new}} - \bar{Y}_{\text{old}}
            \approx 1.69.
        \]

        \item Permutation test:
        \[
            p\text{-value} \approx 0.
        \]

        \item \textbf{Conclusion:} Reject \(H_0\).  
        The new landing page significantly increases time on page.
    \end{itemize}
\end{frame}

\end{frame}

\begin{frame}{Time Spent: Variance Ratio Test}
    \begin{itemize}
        \item Hypotheses:
        \[
            H_0: \sigma^2_{\text{new}} \le \sigma^2_{\text{old}}, \qquad
            H_1: \sigma^2_{\text{new}} > \sigma^2_{\text{old}}.
        \]

        \item Sample variances:
        \[
            s^2_{\text{old}} \approx 6.667, \qquad
            s^2_{\text{new}} \approx 3.302.
        \]

        \item Observed ratio:
        \[
            t_{\text{obs}}
            = \frac{s^2_{\text{new}}}{s^2_{\text{old}}}
            \approx 0.495.
        \]

        \item Permutation test:
        \[
            p\text{-value} \approx 0.995.
        \]

        \item \textbf{Conclusion:} Fail to reject \(H_0\).  
        Variance on the new page is smaller (more consistent behavior).
    \end{itemize}
\end{frame}

\section{Results: Conversion Rates}
% =========================================

\begin{frame}{Conversion Rate: Difference in Means}
    \begin{itemize}
        \item Binary conversion variable:
        \[
        \text{converted\_num} =
        \begin{cases}
            1 & \text{if converted}, \\
            0 & \text{otherwise}.
        \end{cases}
        \]

        \item Hypotheses:
        \[
            H_0: p_{\text{new}} \le p_{\text{old}}, \qquad
            H_1: p_{\text{new}} > p_{\text{old}}.
        \]

        \item Sample conversion rates:
        \[
            p_{\text{old}} \approx 0.42, \qquad
            p_{\text{new}} \approx 0.66.
        \]

        \item Observed difference:
        \[
            t_{\text{obs}}
            = p_{\text{new}} - p_{\text{old}}
            \approx 0.24.
        \]

        \item Permutation test:
        \[
            p\text{-value} \approx 0.0042.
        \]

        \item \textbf{Conclusion:} Reject \(H_0\).  
        The new page significantly improves conversion rate.
    \end{itemize}
\end{frame}


\begin{frame}{Conversion Rate: Variance Ratio Test}
    \begin{itemize}
        \item Hypotheses:
        \[
            H_0: \sigma^2_{\text{new}} \le \sigma^2_{\text{old}}, \qquad
            H_1: \sigma^2_{\text{new}} > \sigma^2_{\text{old}}.
        \]

        \item Sample variances:
        \[
            s^2_{\text{old}} \approx 0.248, \qquad
            s^2_{\text{new}} \approx 0.228.
        \]

        \item Observed ratio:
        \[
            t_{\text{obs}} =
            \frac{s^2_{\text{new}}}{s^2_{\text{old}}}
            \approx 0.92.
        \]

        \item Permutation test:
        \[
            p\text{-value} \approx 0.701.
        \]

        \item \textbf{Conclusion:}  
        Fail to reject \(H_0\).  
        Variability in conversion rate does not differ between pages.
    \end{itemize}
\end{frame}

\section{Discussion and Limitations}
% =========================================

\begin{frame}{Discussion of Findings}
    \textbf{Time on Page}
    \begin{itemize}
        \item Strongly significant mean difference (\(p \approx 0\)).
        \item New page increases average engagement by \(\approx 1.69\) minutes.
        \item Variance lower on new page â†’ more consistent behavior.
    \end{itemize}

    \vspace{0.2cm}

    \textbf{Conversion Rate}
    \begin{itemize}
        \item Mean difference significant (\(p \approx 0.0042\)).
        \item New page increases conversion by \(\approx 24\%\).
        \item No evidence of variance change (\(p \approx 0.70\)).
    \end{itemize}
\end{frame}


\begin{frame}{Limitations of Permutation Tests}
    \begin{itemize}
        \item \textbf{Assumption of identical distributions under \(H_0\):}
        Permutation tests require equal distributions under the null;  
        unequal variances may reduce exactness.

        \item \textbf{No confidence intervals:}  
        Permutation tests provide p-values but not effect-size CIs.

        \item \textbf{Sensitivity to distribution shape:}  
        Skewness or variance differences can affect results even when testing the mean.

        \item \textbf{Computational cost:}  
        Large \(R\) improves stability but increases runtime.
    \end{itemize}
\end{frame}
