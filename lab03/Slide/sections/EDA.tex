\section{Simulation Mini-Project}

\begin{frame}[plain]

    \begin{center}
        \color{blue}\Huge\textbf{Simulation Mini-Project} \\
        \vspace{0.5cm}
        \Large AB test to determine the effectiveness of a new landing page
    \end{center}
\end{frame}

%--------------- slide 1 ----------------
\begin{frame}
    \frametitle{Context \& Experiment Design}
    \begin{itemize}
        \item \textbf{Context:} The Design team developed a \textbf{new landing page} (updated layout, more relevant content) to attract new subscribers.
        
        \item \textbf{Objective:} To evaluate the \textbf{effectiveness} of this redesign compared to the original version.
        
        \item \textbf{A/B Test Design:}
        \begin{itemize}
            \item \textbf{Total Users:} 100 users were randomly selected.
            \item \textbf{Group Division:} 
            \begin{enumerate}
                \item \textbf{Control Group:} Shown the existing page (Old Page).
                \item \textbf{Treatment Group:} Shown the new version (New Page).
            \end{enumerate}
        \end{itemize}
        
        \item \textbf{Data Collection:} User interaction data from both groups was collected and analyzed.
    \end{itemize}

\end{frame}

%--------------- slide 2 ----------------

\begin{frame}

    \frametitle{E-news Express Analysis Objectives}
    As a Data Scientist at E-news Express, we need to determine the effectiveness of the new landing page by answering two main questions:
    
    \vspace{0.3cm}
    
    \begin{enumerate}
        \item \textbf{Engagement Time:}
        \begin{itemize}
            \item Do users \textbf{spend more time} on the new landing page than on the existing page?
            \item \textit{Relevant Metric:} Time spent on the page (minutes).
        \end{itemize}
        
        \pause 
        
        \vspace{0.5cm}
        
        \item \textbf{Conversion Rate:}
        \begin{itemize}
            \item Is the conversion rate (proportion of users who subscribe) for the new page \textbf{greater than} the conversion rate for the old page?
            \item \textit{Relevant Metric:} Converted (Binary: Yes/No).
        \end{itemize}
    \end{enumerate}

\end{frame}

%--------------- slide 3 ----------------

\begin{frame}
    \frametitle{Dataset Structure}
    \textbf{The dataset includes 6 main columns:}
    
    \vspace{0.3cm}
    
    \begin{tabular}{|l|p{6cm}|}
        \hline
        \textbf{Column} & \textbf{Description} \\
        \hline
        \texttt{user\_id} & Unique identifier for the user. \\
        \hline
        \texttt{group} &  Whether the user belongs to the first group (control) or the second group (treatment) \\
        \hline
        \texttt{landing\_page} & Which page they interacted with (\texttt{old}/\texttt{new}). \\
        \hline
        \texttt{time\_spent\_on\_the\_page} & Time (in minutes) spent by the user on the landing page. \\
        \hline
        \texttt{converted} & Binary variable: Whether the user \textbf{subscribed}. \\
        \hline
        \texttt{language\_preferred} & Language chosen by the user to view the landing page. \\
        \hline
    \end{tabular}
    
    \vspace{0.3cm}
    
\end{frame}
